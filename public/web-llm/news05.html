async function myFetchFeed(feedUrl, apiKey) { try { let response; if (apiKey !== '') { response = await fetch(feedUrl + apiKey); } else { response = await fetch(feedUrl); } const data = await response.json(); if (data.articles && data.articles.length > 0) { return data.articles[0].title; } else { return "No news available."; } } catch (error) { return "Error fetching news."; } }
